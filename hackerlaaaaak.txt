//
#include "stdafx.h"
#include <stdio.h>
#include <iostream>
using namespace std;
void main(void)                      /* ejemplo de uso de un puntero */
{
int index,*pt1,*pt2;

   index = 39;                      /* cualquier valor numérico */
   pt1 = &index;                   /* la dirección de index */
   pt2 = pt1;
   printf("El valor es  %d %d %d\n",index,*pt1,*pt2);
   *pt1 = 13;           /* esto cambia el valor de index */
   
   cout << &index << endl;
   cout << "SIN NADA " << pt1 << endl;
   cout << "SIN NADA " << pt2 << endl;
   cout << "* "<<*pt1 << endl;
   cout << "* "<<*pt2 << endl;
   cout << "& " << &pt1 << endl;
   cout << "& " << &pt2 << endl;

   printf("El valor es %d %d %d\n",index,*pt1,*pt2);
   system("pause");
}
